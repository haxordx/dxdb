submitUpdate = function(anEntry, anUpdate) {
  pubnub.publish({
    channel : theChannel,
    message : {'entry' : anEntry, 'update' : anUpdate}
  },
  function(status, response) {
    if (status.error) {
      console.log(status)
    }
    else {
      displayMessage('[PUBLISH: sent]',
        'timetoken: ' + response.timetoken);

      updateText.value="";
    }
  });
};
